CREATE DATABASE IF NOT EXISTS studysync;
USE studysync;

CREATE TABLE users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  full_name VARCHAR(100) NOT NULL,
  email VARCHAR(150) NOT NULL UNIQUE,
  password_hash VARCHAR(255) NOT NULL,
  dob DATE DEFAULT NULL,
  daily_capacity INT DEFAULT 4,
  xp INT DEFAULT 0,
  level INT DEFAULT 1,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE tasks (
  id INT PRIMARY KEY AUTO_INCREMENT,
  user_id INT NOT NULL,
  title VARCHAR(200) NOT NULL,
  description TEXT,
  subject VARCHAR(100) DEFAULT 'General',
  priority ENUM('High','Medium','Low') DEFAULT 'Medium',
  task_date DATE,
  task_time TIME DEFAULT '00:00:00',
  estimated_time INT DEFAULT 60,
  is_urgent TINYINT(1) DEFAULT 0,
  is_important TINYINT(1) DEFAULT 0,
  reminder TINYINT(1) DEFAULT 0,
  color VARCHAR(20) DEFAULT '#00B4BE',
  status ENUM('pending','completed') DEFAULT 'pending',
  smart_score INT DEFAULT 0,
  order_number INT DEFAULT 0,
  completed_at TIMESTAMP NULL,
  time_spent_mins INT DEFAULT 0,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE notes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    title VARCHAR(255) NOT NULL,
    content TEXT,
    category ENUM('Study', 'Personal', 'Social', 'Important') DEFAULT 'Study',
    is_starred TINYINT(1) DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE friends (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    friend_id INT NOT NULL,
    status ENUM('pending', 'accepted') DEFAULT 'pending',
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (friend_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE daily_activity (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    activity_date DATE NOT NULL,
    UNIQUE KEY uq_user_day (user_id, activity_date),
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

function getLocalDateString(date) {
    let d = new Date(date);
    return `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}-${String(d.getDate()).padStart(2, '0')}`;
}

async function generateVirtualSchedule(user, tasks) {
    const dailyCapacityMins = (parseInt(user.daily_capacity) || 4) * 60;
    let pool = tasks.filter(t => t.status === 'pending').map(t => ({
        ...t, remainingMins: parseInt(t.estimated_time) || 60 
    }));

    let fixed = pool.filter(t => t.task_time !== '00:00:00');
    let floating = pool.filter(t => t.task_time === '00:00:00');

    floating.sort((a, b) => (a.order_number - b.order_number) || (b.smart_score - a.smart_score));

    let schedule = {};
    let simDate = new Date();
    for (let i = 0; i < 30; i++) {
        const dateKey = getLocalDateString(simDate);
        schedule[dateKey] = [];
        let used = 0;

        fixed.filter(t => t.task_date === dateKey).forEach(t => {
            schedule[dateKey].push({...t, isFixed: true, segmentMins: t.remainingMins});
            used += t.remainingMins;
            t.remainingMins = 0;
        });

        for (let t of floating) {
            if (t.remainingMins <= 0 || t.task_date > dateKey) continue;
            let space = dailyCapacityMins - used;
            if (space <= 0) break;
            let assign = Math.min(t.remainingMins, space);
            schedule[dateKey].push({...t, isFixed: false, segmentMins: assign});
            t.remainingMins -= assign;
            used += assign;
        }
        simDate.setDate(simDate.getDate() + 1);
    }
    return schedule;
}
